Programação Assíncrona em JavaScript

Programação assíncrona é essencial para lidar com operações que podem levar algum tempo, como requisições de rede. Em JavaScript, podemos usar funções assíncronas e a palavra-chave await para trabalhar de maneira eficiente com tais operações.

Exemplo de Código

        
        document.addEventListener("DOMContentLoaded", () => {
            // Obtendo referências para os elementos HTML
            const fetchDataButton = document.getElementById("fetchDataButton");
            const outputDiv = document.getElementById("output");

            // Função de callback para processar os dados obtidos
            const handleDataCallback = (error, data) => {
                if (error) {
                    console.error("Erro ao buscar dados:", error);
                    outputDiv.innerHTML = "
Ocorreu um erro ao buscar os dados
";
                } else {
                    // Exibindo os resultados no HTML
                    outputDiv.innerHTML = `
                        
UserID: ${data.userId}

                        
ID: ${data.id}

                        
Title: ${data.title}

                        
Completed: ${data.completed}

                    `;
                }
            };

            // Função assíncrona que utiliza uma função de callback
            const fetchDataWithCallback = (callback) => {
                // Realizando uma requisição assíncrona usando fetch
                fetch("https://jsonplaceholder.typicode.com/todos/1")
                    .then((response) => response.json())
                    .then((data) => callback(null, data))
                    .catch((error) => callback(error, null));
            };

            // Adicionando um evento de clique ao botão
            fetchDataButton.addEventListener("click", () => {
                // Chamada da função assíncrona com callback
                fetchDataWithCallback(handleDataCallback);
            });
        });
